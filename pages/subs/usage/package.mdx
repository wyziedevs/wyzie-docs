## Usage

<Tabs items={["Fetching Subtitles", "VTT Parser", "Get Sources"]}>
  <Tabs.Tab>
    ```ts
    import { type SubtitleData, searchSubtitles, parseToVTT } from "wyzie-lib";


    const data: SubtitleData[] = await searchSubtitles({
      tmdb_id: 286217,
      format: ["srt"],
      language: ["en"],
      release: ["web-dl", "1080p"],
      origin: ["WEB"],
      key: "YOUR_KEY",
    });

    console.log(data[0].url);
    ```
  </Tabs.Tab>
  <Tabs.Tab>
    ```ts
    import { type SubtitleData, searchSubtitles, parseToVTT } from "wyzie-lib";


    const data: SubtitleData[] = await searchSubtitles({ tmdb_id: 286217, format: "srt", key: "YOUR_KEY" });

    const vttContent = await parseToVTT(data[0].url);
    console.log(vttContent);
    ```

  </Tabs.Tab>
  <Tabs.Tab>
    ```ts
    import { getSources } from "wyzie-lib";


    const sources = await getSources();
    console.log(sources); // ["subdl", "subf2m", "opensubtitles", "podnapisi", "animetosho", "gestdown", "jimaku", "kitsunekko", "yify", "ajatttools"]
    ```

  </Tabs.Tab>
</Tabs>

### Configuration

<TP id="subs.pkg.config.p1" />

```ts
import { configure, searchSubtitles, parseToVTT } from 'wyzie-lib';


configure({ 
  baseUrl: 'https://my-custom-subtitle-scraper.com',
  key: 'YOUR_KEY',
});


const subtitles = await searchSubtitles({
  tmdb_id: 123456,
  language: "en"
});
```
